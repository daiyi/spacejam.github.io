GCounter is a grow-only witnessed counter.

Examples

use crdts::GCounter;
let (mut a, mut b) = (GCounter::new(), GCounter::new());
a.increment("A".to_string());
b.increment("B".to_string());
assert_eq!(a.value(), b.value());
assert!(a == b);
a.increment("A".to_string());
assert!(a > b);
