Safety

add should never be passed identical actor arguments for different replicas. This will result in data loss:

use crdts::Orswot;
let (mut a, mut b) = (Orswot::new(), Orswot::new());
a.add(1, 1);
b.add(2, 1);
a.merge(b);
assert!(a.value().is_empty());
